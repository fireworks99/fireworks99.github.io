function Ak = nafit1( A, x, xk, yk, m)
Pni=inv(P(x,m));
phik2phik=phi(xk,m);
phi2y=phiy(xk, yk, m);
Ak=A'-Pni*phik2phik*A'+Pni*phi2y;
Ak=fliplr(Ak');
end

function phi2y = phiy(xk, yk, m)
phi2y = zeros(m+1, 1);
for i=0:m
    phi2y(i+1)=xk^i*yk;
end
end

function phik2phik = phi(xk, m)
phik2phik = zeros(m+1, m+1);
for i=0:m
    for j=0:m
        phik2phik(i+1,j+1)=xk^(i+j);
    end
end
end

function Pres = P(x, m)
len = length(x);
Pres = zeros(m+1, m+1);
for k=1:len
    Pres = Pres + phi(x(k), m);
end
end
