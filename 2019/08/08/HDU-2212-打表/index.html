ll fac[20];

void get_fac()
{
    fac[0] = 1;
    for(int i = 1; i <= 12; ++i)
        fac[i] = fac[i - 1] * i;
}

bool check(ll n)
{
    int tem = n;
    ll res = 0;
    while(tem)
    {
        res += fac[tem % 10];
        tem /= 10;
    }
    if(res == n)
        return 1;
    return 0;
}

int main()
{
    freopen("00out.txt", "w", stdout);
    get_fac();
    for(ll i = 1; i <= 2147483647; ++i)
        if(check(i))
            cout << i << ',';
    return 0;
}
