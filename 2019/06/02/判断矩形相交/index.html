判断矩形相交

int check(int x1,int y1,int x2,int y2,int x3, int y3,int x4, int y4)
{
    if(x2 < x3 || x4 < x1 || y4 < y1 || y2 < y3)
        return 0;
    return (min(y4, y2)- max(y1,y3)) * (min(x4,x2) - max(x1, x3));
}
