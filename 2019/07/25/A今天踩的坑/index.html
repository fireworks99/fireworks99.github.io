__builtin_popcount()计算一个十进制数的二进制形式里有几个1

#include <cstdio>
#include <string>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;


int main()
{
    char s[50];
    scanf("%s", s);
    long a = strtol(s, NULL, 2);
    cout << a << '\n';
    int num = __builtin_popcount(a);
    cout << num << '\n';
}
/// s : 1101
/// s : 13
/// s : 3
